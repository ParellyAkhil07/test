
    // Method to decode the string using an iterative approach
    public static String decode(String s) {
        // Initialize an empty string to store the decoded result
        String decoded = "";

        // Iterate over each character in the string
        for (int l = 0; l < s.length(); l++) {
            try {
                // Try to parse the current character as an integer
                int skip = Integer.parseInt(String.valueOf(s.charAt(l)));
                // Append the character found after skipping the specified number of characters
                decoded += s.charAt(skip + l + 1);
            } catch (NumberFormatException e) {
                // Continue if the current character is not a number
                continue;
            }
        }

        // Return the decoded string
        return decoded;
    }

    // Method to decode the string using a recursive approach
    public static String decode2(String s) {
        // Call the recursive helper function with initial parameters
        return _decode2(s, "", 0);
    }

    // Recursive helper function for decode2
    private static String _decode2(String s, String decoded, int i) {
        // Base case: if the index exceeds the string length, return the decoded result
        if (i >= s.length()) {
            return decoded;
        }

        // Get the current character
        char current = s.charAt(i);

        // Check if the current character is a digit
        if (Character.isDigit(current)) {
            // Append the character found after skipping the specified number of characters
            decoded += s.charAt(Character.getNumericValue(current) + i + 1);
        }

        // Recursive call with the next index
        return _decode2(s, decoded, i + 1);
    }

    public static void main(String[] args) {
        // Test cases
        System.out.println(decode("0h")); // Output: h
        System.out.println(decode("2abh")); // Output: h
        System.out.println(decode("0h1ae2bcy")); // Output: hey

        System.out.println(decode2("0h")); // Output: h
        System.out.println(decode2("2abh")); // Output: h
        System.out.println(decode2("0h1ae2bcy")); // Output: hey
    }
