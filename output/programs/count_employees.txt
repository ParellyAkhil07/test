import java.util.List;

class Node {
    // Node in a tree
    private String name;
    private List<Node> children;

    // Constructor for Node
    public Node(String name, List<Node> children) {
        this.name = name;
        this.children = (children != null) ? children : new ArrayList<>();
    }

    // Method to count employees under this node
    public int countEmployees() {
        // If no children, return 0
        if (children.isEmpty()) {
            return 0;
        }

        int count = 0;

        // Count each child and their employees recursively
        for (Node child : children) {
            count += 1 + child.countEmployees();
        }

        return count;
    }

    public static void main(String[] args) {
        // Test cases
        Node henri = new Node("Henri", null);
        Node nora = new Node("Nora", List.of(henri));
        Node nick = new Node("Nick", null);
        Node janet = new Node("Janet", List.of(nick, nora));
        Node al = new Node("Al", null);
        Node bob = new Node("Bob", null);
        Node jen = new Node("Jen", null);
        Node jessica = new Node("Jessica", List.of(al, bob, jen));
        Node jane = new Node("Jane", List.of(jessica, janet));

        // Output the results of countEmployees for each test case
        System.out.println(henri.countEmployees()); // 0
        System.out.println(janet.countEmployees()); // 3
        System.out.println(jessica.countEmployees()); // 3
        System.out.println(jane.countEmployees()); // 8
    }
